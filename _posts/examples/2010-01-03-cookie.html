Code to implement this

Json data source
{% highlight js%} // json data source var jsonSource = { tree: { id: "root", nodes: [{ id: "master", title: "master node", description: "title of the master link", hasChildren: true, nodes: [{ id: "parent_1", title: "parent 1", description: "title of the parent 1 link", hasChildren: true, nodes: [{ id: "child_1", title: "child 1", description: "title of the child 1 link", hasChildren: false }] },{ id: "parent_2", title: "parent 2", description: "title of parent 2 link", hasChildren: false }] }] } }; {% endhighlight %}
initialisation
{% highlight js%} var tree = Vtree.create({ container: jQuery("#treeContainer"), dataSource: jsonSource, initiallyOpen: ["master", "parent_1"], //cookie preferences have priority over this setting initiallyChecked: ["parent_1"], //cookie preferences have priority over this setting plugins:["checkbox", "cookie"] }); {% endhighlight %}
