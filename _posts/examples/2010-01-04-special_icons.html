Code to implement this

Json data source
{% highlight js%} // json data source var jsonSource = { tree: { id: "iconsTree", nodes: [{ id: "master", title: "master node", customClass: "title", // make the node like a header iconClass:"default", // a default style for icon integrated in css hasChildren: true, nodes: [{ id: "parent_1", title: "parent 1", iconPath: { //specifying here 2 icon urls for different state of node open: "/vtree/img/icon_folder_open_16.png", close: "/vtree/img/icon_folder_closed_16.png" }, hasChildren: true, nodes: [{ id: "child_1", title: "child 1", iconPath: "/vtree/img/file.png", //specifying only one icon url hasChildren: false }] },{ id: "parent_2", title: "parent 2", iconClass: "customFolder", // a default folder icon style integrated in css hasChildren: true, nodes: [{ id: "child_2", title: "child 2", iconClass: "customFile", // a default file icon style integrated in css hasChildren: false }] }] }] } }; {% endhighlight %}
initialisation
{% highlight js%} var tree = Vtree.create({ container: jQuery("#treeContainerIcons"), dataSource: jsonSource initially_open: ["parent_1", "parent_2"] }); {% endhighlight %}
